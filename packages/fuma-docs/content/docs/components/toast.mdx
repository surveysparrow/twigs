## Import

```jsx
import { Toastr, toast } from "@sparrowengg/twigs-react";
```

## Usage

```jsx
const ToastExample = () => {
  return (
    <>
      <Toastr duration={10000} />
      <Button
        variant="outline"
        onClick={() => {
          toast({
            variant: "success",
            title: "Success!",
            description: "Your action was completed successfully."
          });
        }}
      >
        Show Toast
      </Button>
    </>
  );
};
```
