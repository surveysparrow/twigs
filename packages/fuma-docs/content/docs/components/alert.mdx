## Import

```jsx
import { Alert, AlertDescription } from "@sparrowengg/twigs-react";
```

## Usage

```jsx
<Alert status="info">
  <AlertDescription>
    Password should have a minimum length of 6.
  </AlertDescription>
</Alert>
```
