## Import

```jsx
import { Select } from "@sparrowengg/twigs-react";
```

## Usage

```jsx
const SelectExample = () => {
  const [selectedValue, setSelectedValue] = useState(null);

  return (
    <Select
      placeholder="Choose an option..."
      options={options}
      value={selectedValue}
      onChange={setSelectedValue}
      size="md"
      variant="default"
    />
  );
};
```
