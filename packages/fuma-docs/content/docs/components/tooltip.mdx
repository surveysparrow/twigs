## Import

```jsx
import { TooltipProvider, Tooltip } from "@sparrowengg/twigs-react";
```

## Usage

```jsx
const TooltipExample = () => {
  return (
    <TooltipProvider delayDuration={0}>
      <Tooltip content="This is a tooltip">
        <Button>Hover me</Button>
      </Tooltip>
    </TooltipProvider>
  );
};
```
